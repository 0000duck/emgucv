FIND_PROGRAM(MSBUILD_PROGRAM
  NAMES MSBuild
  HINTS
  "$ENV{SYSTEMROOT}/Microsoft.NET/Framework/v3.5"
  "C:/WINDOWS/Microsoft.NET/Framework/v3.5"
)

IF(${MSBUILD_PROGRAM} MATCHES "MSBuild") 
MESSAGE(STATUS "MSBUILD_PROGRAM: ${MSBUILD_PROGRAM}")
PROJECT(Emgu.CV.Document.Html)
ADD_CUSTOM_TARGET(
   ${PROJECT_NAME}
   COMMAND ${MSBUILD_PROGRAM} ${CMAKE_CURRENT_SOURCE_DIR}/Emgu.CV.html.shfbproj
)
ADD_DEPENDENCIES(${PROJECT_NAME} Emgu.Util Emgu.CV Emgu.CV.UI Emgu.CV.ML)

PROJECT(Emgu.CV.Document.Mshelp)
ADD_CUSTOM_TARGET(
   ${PROJECT_NAME}
   COMMAND ${MSBUILD_PROGRAM} ${CMAKE_CURRENT_SOURCE_DIR}/Emgu.CV.mshelp.shfbproj
)
ADD_DEPENDENCIES(${PROJECT_NAME} Emgu.Util Emgu.CV Emgu.CV.UI Emgu.CV.ML)
ELSE()
MESSAGE(STATUS "MSBUILD_PROGRAM: NOT FOUND")
ENDIF()
