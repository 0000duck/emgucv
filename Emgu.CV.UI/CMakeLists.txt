IF(EMGU_CV_BUILD OR EMGU_CV_EXAMPLE_BUILD)

PROJECT(Emgu.CV.UI)

SET(SRC_CV_UI *.cs Properties/*.cs)

GENERATE_DOCUMENT(${LIBRARY_OUTPUT_PATH}/${PROJECT_NAME})
ADD_CS_REFERENCES("${LIBRARY_OUTPUT_PATH}/Emgu.Util.dll;${LIBRARY_OUTPUT_PATH}/Emgu.CV.dll;System.Drawing.dll;System.Data.dll;System.Windows.Forms.dll;${LIBRARY_3RDPARTY_PATH}/ZedGraph.dll")
SIGN_ASSEMBLY(CV.UI.snk)

FILE(GLOB RESX_SRC 
   "${CMAKE_CURRENT_SOURCE_DIR}/*.resx" 
   "${CMAKE_CURRENT_SOURCE_DIR}/Properties/*.resx")
FOREACH (RESX ${RESX_SRC})
   STRING(REPLACE "${CMAKE_CURRENT_SOURCE_DIR}/" "${PROJECT_NAME}." RESOURCES_NAME ${RESX})
   STRING(REPLACE ".resx" ".resources" RESOURCES_NAME ${RESOURCES_NAME})
   STRING(REPLACE "/" "." RESOURCES_NAME ${RESOURCES_NAME})
   ADD_CS_RESOURCES(${RESX} ${RESOURCES_NAME})
ENDFOREACH(RESX)

ADD_CS_FILE_TO_DEPLOY(${CMAKE_CURRENT_SOURCE_DIR}/../lib/3rdParty/ZedGraph.dll)
ADD_CS_LIBRARY(${PROJECT_NAME} "${SRC_CV_UI}" ALL)

ADD_DEPENDENCIES(${PROJECT_NAME} Emgu.CV Emgu.Util)
INSTALL(FILES ${LIBRARY_OUTPUT_PATH}/${PROJECT_NAME}.dll DESTINATION bin
COMPONENT binary)

ENDIF()
